% -------------------- OBJETO -------------------------------
\section{Objeto}
\accentbar

TDF producirá y fijará una sesión musical en vivo (la “\textbf{Sesión}”) en \textbf{TDF Estudio (Quito) — sin público —} compuesta por:
\begin{enumerate}[leftmargin=*,label=\alph*)]
  \item performance de hasta \textbf{30 minutos};
  \item mini‑entrevista y material \textit{behind‑the‑scenes};
  \item captación de audio multipista y video 4K multicámara.
\end{enumerate}
Las fechas se coordinarán con preferencia por \textbf{sábados} como día base de grabación y \textbf{estreno en domingo}. Cualquier locación externa requerirá \textbf{Addendum} con estándar técnico TDF y equipo TDF al mando.

% -------------------- PRODUCCIÓN Y TÉCNICA -----------------
\section{Producción y técnica}
\accentbar
\begin{enumerate}[leftmargin=*,label=\arabic*.]
  \item \textbf{Rider:} el Artista entregará \textbf{rider técnico}, \textbf{setlist} y \textbf{input list} por canción (con BPM, tonalidad y gear preferido del inventario) con al menos \textbf{48 horas} de antelación. Los cambios posteriores se aceptan solo si producción confirma viabilidad.
  \item \textbf{Backline:} TDF proveerá batería, amplificadores y teclado estándar, salvo pacto distinto por escrito. Si se usan instrumentos del Artista, deben llegar ajustados y afinados.
  \item \textbf{Ambiente:} no habrá captación de “stems de público” al ser una Sesión sin audiencia.
  \item \textbf{Normas de set:} aplican las del \textbf{Anexo I}.
  \item \textbf{Créditos de producción (para títulos y descripciones):}\\
        Video / Dirección de fotografía: \textbf{Francisco Almeida}.\\
        Edición: \textbf{Lucía Orbea Bastl}.\\
        Producción visual / dirección de arte: \textbf{Claudia Palma}.\\
        Community Management: \textbf{Camila Maldonado} (únicamente CM).
  \item \textbf{Staff técnico:} TDF decide asignación de ingenieros, operadores y asistentes. El Artista podrá sugerir referencias de mezcla, no personal técnico.
  \item \textbf{Re‑takes:} hasta \textbf{2 tomas completas} por canción cuando el horario lo permita; tomas adicionales quedan a criterio de producción.
\end{enumerate}

% -------------------- DERECHOS Y CRÉDITOS ------------------
\section{Derechos, cesión y créditos}
\accentbar
\begin{enumerate}[leftmargin=*,label=\arabic*.]
  \item El Artista \textbf{cede a TDF}, con carácter \textbf{exclusivo, perpetuo y mundial}, todos los derechos patrimoniales de fijación, reproducción, distribución, comunicación pública, transformación, sincronización y cualesquiera otros sobre los másters de audio y audiovisual de la Sesión, en cualquier formato presente o futuro (incluyendo VR y metaverso).
  \item La \textbf{propiedad de los másters} de audio y audiovisual será de \textbf{TDF}.
  \item \textbf{Créditos:} TDF acreditará al Artista como “\ArtistaNombre — TDF Live Sessions” en metadatos, descripción y miniaturas.
  \item \textbf{First‑release:} el Artista no publicará por su cuenta material de la Sesión durante \textbf{\FirstReleaseDias\ días} contados desde el estreno oficial por TDF.
  \item \textbf{Obras originales:} el repertorio será \textbf{sólo obras originales} o debidamente licenciadas; el Artista garantiza contar con los derechos necesarios para fijar e interpretar.
  \item \textbf{Uso promocional:} TDF podrá utilizar fragmentos, stems y capturas de backstage para promoción cruzada en redes, medios y pauta digital.
\end{enumerate}

% -------------------- MODELO ECONÓMICO ---------------------
\section{Modelo económico, contabilidad y pagos}
\accentbar
\begin{enumerate}[leftmargin=*,label=\arabic*.]
  \item \textbf{Recuperación de costos:} los ingresos brutos asociados a la Sesión (YouTube/Adsense, DSPs, patrocinios in‑video, sincronizaciones y reventas de NFT si aplicaran) se destinarán primero a \textbf{recuperar costos} de producción detallados en el \textbf{Anexo II}, con \textbf{tope máximo USD \TopeRecoup}.
  \item \textbf{Split:} una vez recuperados los costos, las utilidades netas se repartirán \textbf{50\% TDF / 50\% Artista}.
  \item \textbf{Liquidación:} TDF rendirá cuentas con \textbf{periodicidad anual}, en USD, pagando dentro de \textbf{10 días} tras el cierre del período.
  \item \textbf{NFT reventa (si existiere):} \textbf{royalty 10\%} compartido 50/50.
\end{enumerate}

% -------------------- ENTREGABLES Y REVISIONES -------------
\section{Entregables y revisiones}
\accentbar
\begin{enumerate}[leftmargin=*,label=\arabic*.]
  \item \textbf{Entregables:} este Contrato \textbf{no} establece obligación de entrega de materiales al Artista (video, audio o stems). Cualquier entrega será a \textbf{discreción de TDF}.
  \item \textbf{Escucha crítica:} se realizará una \textbf{sesión de escucha} en el estudio; las observaciones deberán presentarse \textbf{durante} dicha sesión.
  \item \textbf{Postproducción y QC (objetivos):} \textbf{edición 48h}, \textbf{mezcla 48h}, \textbf{master/color/QC 48h}. \textbf{QC: Audio (TDF)}.
  \item \textbf{Embargo:} el Artista no publicará material de la Sesión antes del estreno sin autorización de TDF.
  \item \textbf{Promoción (best effort):} el Artista hará su mejor esfuerzo por publicar \textbf{10} piezas (previo y posterior) etiquetando @tdfrecords.
  \item \textbf{Respaldo de multitracks:} TDF conservará los proyectos y multitracks \textbf{90 días} después del estreno; no se obliga a retenerlos más allá de ese plazo.
\end{enumerate}

% -------------------- CANCELACIONES / SEGUROS --------------
\section{Cancelaciones, seguros y fuerza mayor}
\accentbar
\begin{enumerate}[leftmargin=*,label=\arabic*.]
  \item \textbf{Cancelación del Artista:} si cancela con menos de \textbf{72 horas}, reembolsará a TDF el \textbf{50\%} de costos directos no recuperables.
  \item \textbf{Cancelación de TDF:} si cancela sin fuerza mayor, cubrirá traslados comprobados del Artista y reagendará en un máximo de \textbf{60 días}.
  \item \textbf{Fuerza mayor:} sismo, corte eléctrico, orden sanitaria u otros eventos fuera de control eximen de responsabilidad.
  \item \textbf{Seguros e instrumentos:} TDF mantiene RC; en caso de robo por negligencia de TDF, cobertura hasta \textbf{USD 2\,000} por instrumento o su valor de mercado (lo menor).
\end{enumerate}

\section*{ANEXO I — Normas de set y seguridad}
\accentbar
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Sin público}; acceso restringido a equipo y Artista.
